%%%%%%%%%%%%%%%%%% LISTA DE SÍMBOLOS %%%%%%%%%%%%%%%%%%
% ================= ESTILOS =================
% Estilo de vector y estilo de matriz
\newcommand{\estVect}[1]{\mathbf{#1}}
\newcommand{\estMatr}[1]{\mathbf{#1}}

% ================= LETRAS =================
\def \nn {\mathcal{N}}
\def \mm {\mathcal{M}}
\def \dd {\mathcal{D}}

% ================= DISTRIBUCIONES =================
% pdf de la Normal
\def \pdfN#1#2{\nn (#1,\, #2)}
\def \pN#1#2#3{\nn (#1 \, |\, #2,\, #3)}

% ================= ÁLGEBRA LINEAL =================
% Matriz identidad
\def \MI {\estMatr{I}}
% X matrix
\def \MX {\estMatr{X}}
% S matrix
\def \MS {\estMatr{S}}

% x vector
\def \vx {\estVect{x}}
% y vector
\def \vy {\estVect{y}}
% w vector
\def \vw {\estVect{w}}
% m vector
\def \vm {\estVect{m}}
% cero vector
\def \vcero {\estVect{0}}
